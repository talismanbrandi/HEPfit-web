65 double NPHiggs::obliqueS() const
66 {
67  double Lambda;
68  if (LambdaNP_in != 0.0)
69  Lambda = LambdaNP_in;

71  Lambda = pow(10.0, 19.0);
72  else
73  Lambda = 4.0 * M_PI * v() / sqrt(fabs(1.0 - a * a));
74
