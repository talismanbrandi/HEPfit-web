96 double HiggsKvgenKf::muVBFpVH(const double sqrt_s) const
97 {
98  double sigmaWH_SM = trueSM.computeSigmaWH(sqrt_s);
99  double sigmaZH_SM = trueSM.computeSigmaZH(sqrt_s);
100  double sigmaWF_SM = trueSM.computeSigmaWF(sqrt_s);
101  double sigmaZF_SM = trueSM.computeSigmaZF(sqrt_s);
102  double sigmaZWF_SM = trueSM.computeSigmaZWF(sqrt_s);
103  double sigmaVBF_SM = sigmaWF_SM + sigmaZF_SM + sigmaZWF_SM;
104 
105  double sigmaWH = muWH(sqrt_s) * sigmaWH_SM;
106  double sigmaZH = muZH(sqrt_s) * sigmaZH_SM;
107  double sigmaVBF = muVBF(sqrt_s) * sigmaVBF_SM;
108 
109  return ((sigmaWH + sigmaZH + sigmaVBF) / (sigmaWH_SM + sigmaZH_SM + sigmaVBF_SM));
110 }
