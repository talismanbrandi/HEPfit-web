183 double EW_CHMN::GammaZ_l(QCD::lepton l) const
184 {
185  double Qf = SM.getLeptons(l).getCharge();
186  double gL = gL_l(l);
187  double gR = gR_l(l);
188  double GammaZ = SM.getGF() * pow(SM.getMz(), 3.0) / (6.0 * sqrt(2.0) * M_PI);
189  GammaZ *= ((gL + gR)*(gL + gR) + deltaImKappa_l(l)) * CV_l(l)
190  + (gL - gR)*(gL - gR) * CA_l(l);
191  GammaZ *= 1.0 + 3.0 / 4.0 * Qf * Qf * SM.alphaMz() / M_PI;
192  GammaZ += DeltaEWQCD_l(l);
193  return GammaZ;
194 }
