430 void MonteCarlo::ParseMCMCConfig(std::string file)
431 {
432  std::ifstream ifile(file.c_str());
433  if (!ifile.is_open())
434  throw std::runtime_error("\nERROR: " + MCMCConf + " does not exist. Make sure to specify a valid Monte Carlo configuration file.\n");
435  std::string line;
436  bool IsEOF = false;
437  do {
438  IsEOF = getline(ifile, line).eof();
439  if (*line.rbegin() == '\r') line.erase(line.length() - 1); // for CR+LF
440  if (line.empty() || line.at(0) == '#')
441  continue;
442  boost::char_separator<char> sep(" \t");
443  boost::tokenizer<boost::char_separator<char> > tok(line, sep);
444  boost::tokenizer<boost::char_separator<char> >::iterator beg = tok.begin();
445  if (beg->compare("NChains") == 0) {
446  ++beg;
447  if (isdigit(beg->at(0)) && atoi((*beg).c_str()) > 0) MCEngine.setNChains(atoi((*beg).c_str()));
448  else if (isdigit(beg->at(0)) && atoi((*beg).c_str()) == 0)
449  if (MCEngine.getMPIWorldSize() > 1) MCEngine.setNChains(MCEngine.getMPIWorldSize());
450  else throw std::runtime_error("\nERROR: NChains in the MonteCarlo configuration file: " + MCMCConf + " can only be an integer > 0.\n");
451  else
452  throw std::runtime_error("\nERROR: NChains in the MonteCarlo configuration file: " + MCMCConf + " can only be an integer >= 0.\n");
453  } else if (beg->compare("PrerunMaxIter") == 0) {
454  ++beg;
455  if (isdigit(beg->at(0)) && atoi((*beg).c_str()) > 0) MCEngine.SetNIterationsPreRunMax(atoi((*beg).c_str()));
456  else
457  throw std::runtime_error("\nERROR: PrerunMaxIter in the MonteCarlo configuration file: " + MCMCConf + " can only be an integer > 0.\n");
458  } else if (beg->compare("NIterationsUpdateMax") == 0) {
459  ++beg;
460  if (isdigit(beg->at(0)) && atoi((*beg).c_str()) > 0) MCEngine.SetNIterationsPreRunCheck(atoi((*beg).c_str()));
461  else
462  throw std::runtime_error("\nERROR: NIterationsUpdateMax in the MonteCarlo configuration file: " + MCMCConf + " can only be an integer > 0.\n");
463  } else if (beg->compare("Seed") == 0) {
464  ++beg;
465  if (!isdigit(beg->at(0))) throw std::runtime_error("\nERROR: Seed in the MonteCarlo configuration file: " + MCMCConf + " can only be a number.\n");
466  int seed = atoi((*beg).c_str());
467  if (seed != 0)
468  MCEngine.SetRandomSeed(seed);
469  } else if (beg->compare("Iterations") == 0) {
470  ++beg;
471  if (isdigit(beg->at(0))) MCEngine.SetNIterationsRun(atoi((*beg).c_str()));
472  else
473  throw std::runtime_error("\nERROR: Iterations in the MonteCarlo configuration file: " + MCMCConf + " can only be an integer.\n");
474  } else if (beg->compare("MinimumEfficiency") == 0) {
475  ++beg;
476  double efficiency = atof((*beg).c_str());
477  if (efficiency > 0. && efficiency <= 1.) MCEngine.SetMinimumEfficiency(efficiency);
478  else
479  throw std::runtime_error("\nERROR: MinimumEfficiency in the MonteCarlo configuration file: " + MCMCConf + " can only be a real number greater than 0.0 and less than or equal to 1.0.\n");
480  } else if (beg->compare("MaximumEfficiency") == 0) {
481  ++beg;
482  double efficiency = atof((*beg).c_str());

634 void MonteCarlo::ReadPreRunData(std::string file)
635 {
636  std::ifstream ifile(file.c_str());
637  if (!ifile.is_open())
638  throw std::runtime_error("\nMonteCarlo::ReadPreRunData ERROR: " + file + " does not exist.\n");
639  std::string line;
640  bool IsEOF = false;
641  std::vector<double> mode;
642  std::vector<double> scale;
643  do {
644  IsEOF = getline(ifile, line).eof();
645  if (line.empty())
646  continue;
647  boost::char_separator<char> sep(" ");
648  boost::tokenizer<boost::char_separator<char> > tok(line, sep);
649  boost::tokenizer<boost::char_separator<char> >::iterator beg = tok.begin();
650  ++beg;
651  mode.push_back(atof((*beg).c_str()));
652  ++beg;
653  scale.push_back(atof((*beg).c_str()));
654  } while (!IsEOF);
655  if (mode.size() != MCEngine.GetNParameters())
656  throw std::runtime_error("\nMonteCarlo::ReadPreRunData ERROR: wrong data size.\n");
657  std::vector<double> mode_all;
658  std::vector<double> scale_all;
659  for (unsigned int i = 0; i < MCEngine.GetNChains(); i++){
660  mode_all.insert(mode_all.end(), mode.begin(), mode.end());
661  scale_all.insert(scale_all.end(), scale.begin(), scale.end());
662  }
663  MCEngine.SetInitialPositions(mode_all);
664  MCEngine.SetInitialScaleFactors(scale_all);
665 }
