145 void MFV::SetSoftTerms()
146 {
147  // Colangelo's expressions in Colangelo's basis
148  X.Update(myCKM);
149  msQhat2 = gslpp::matrix<gslpp::complex>::Id(3) * a1 + X.GetX13() * x1 + X.GetX1() * y1 +
150  X.GetX5() * y2 + X.GetX9() * y2.conjugate();
151  msUhat2 = gslpp::matrix<gslpp::complex>::Id(3) * a2 + X.GetX1() * x2;
152  msDhat2 = gslpp::matrix<gslpp::complex>::Id(3) * a3 + X.GetX1() * y3 + X.GetX3() * w1 +
153  X.GetX4() * w1.conjugate();
154  TUhat = X.GetX5() * a4 + X.GetX1() * y4 + X.GetX6() * w2;
155  TDhat = X.GetX1() * a5 + X.GetX5() * y5 + X.GetX2() * w3 + X.GetX4() * w4;
156  msLhat2 = gslpp::matrix<gslpp::complex>::Id(3) * a6 + X.GetX1() * y6;
157  msEhat2 = gslpp::matrix<gslpp::complex>::Id(3) * a7 + X.GetX1() * y7;
158  TEhat = X.GetX1() * a8 + X.GetX2() * w5;
159 
160  // rotation to the SCKM basis according to SLHA notation
161 
162  TUhat = sqrt(2.) * TUhat * myCKM.getCKM().hconjugate();
163  TDhat = sqrt(2.) * TDhat;
164  TEhat = sqrt(2.) * TEhat;
165 
166  // msNhat2 and TNhat remain 0.
167 }
