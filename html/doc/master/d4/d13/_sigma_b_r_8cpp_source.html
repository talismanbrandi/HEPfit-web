95 double SigmaBR::muVH(const double sqrt_s) const
96 {
97  double sigmaWH_SM = computeSigmaWH(sqrt_s);
98  double sigmaZH_SM = computeSigmaZH(sqrt_s);
99  double sigmaWH = muWH(sqrt_s) * sigmaWH_SM;
100  double sigmaZH = muZH(sqrt_s) * sigmaZH_SM;
101  double mu = ((sigmaWH + sigmaZH) / (sigmaWH_SM + sigmaZH_SM));
102 
103  return mu;
