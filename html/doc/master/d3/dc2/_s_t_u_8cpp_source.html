25 double STU::F(const double m02, const double m12) const {
26  double F;
27 
28  if(m02 == 0. && m12 != 0.) {
29  F=0.5 * m12;
30  } else if(m02 != 0. && m12 == 0.){
31  F=0.5 * m02;
32  } else if((m02 == 0. && m12 == 0.) || (fabs(m02-m12) < LEPS)){
33  F=0.;
34  } else if (m02 != 0 && m12 != 0){
35  F=0.5 * (m02 + m12) - (m02 * m12) / (m02 - m12) * log(m02 / m12);
36  } else
37  throw std::runtime_error("Error in THDM::F()");
38  return (F);
39 }

85 double DeltaT::computeThValue()
86 {
87  double mHl2=myTHDM->getmHl2();
88  double mHh2=myTHDM->getmHh2();
89  double mA2=myTHDM->getmA2();
90  double mHp2=myTHDM->getmHp2();
91  double sin_ba=myTHDM->getsin_ba();
92  double sin2_ba = sin_ba*sin_ba;
93  double cos2_ba = 1. - sin2_ba;
94  double MZ=myTHDM->getMz();
95  double MZ2 = MZ*MZ;
96  double MW=mycache->MWTHDM(myTHDM->Mw_tree());
97  double MW2 = MW*MW;
98  double s_W2 = 1.0-mycache->cW2THDM(myTHDM->c02());
99 
100  gslpp::complex B0_MZ2_0_MZ2_mHh2;
101  gslpp::complex B0_MZ2_0_MZ2_mHl2;
102  gslpp::complex B0_MZ2_0_MW2_mHh2;
103  gslpp::complex B0_MZ2_0_MW2_mHl2;
104 
105  B0_MZ2_0_MZ2_mHh2 = mycache->B0_MZ2_0_MZ2_mHh2(MZ2,mHh2);
106  B0_MZ2_0_MZ2_mHl2 = mycache->B0_MZ2_0_MZ2_mHl2(MZ2,mHl2);
107  B0_MZ2_0_MW2_mHh2 = mycache->B0_MZ2_0_MW2_mHh2(MZ2,MW2,mHh2);
108  B0_MZ2_0_MW2_mHl2 = mycache->B0_MZ2_0_MW2_mHl2(MZ2,MW2,mHl2);
109 
110  return 1. / 16. / M_PI / MW2 / s_W2 * (F(mHp2,mA2)
111  + sin2_ba * (F(mHp2,mHh2) - F(mA2,mHh2)) + cos2_ba * (F(mHp2,mHl2)
112  - F(mA2,mHl2) + F(MW2,mHh2) - F(MW2,mHl2) - F(MZ2,mHh2)
113  + F(MZ2,mHl2) + 4. * MZ2 * (B0_MZ2_0_MZ2_mHh2.real() - B0_MZ2_0_MZ2_mHl2.real())
114  - 4. * MW2 * (B0_MZ2_0_MW2_mHh2.real() - B0_MZ2_0_MW2_mHl2.real())));
115 }
