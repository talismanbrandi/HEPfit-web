73  continue;
74  } else {

80  // ggH fraction

128  if (covarianceFromConfig) {
129  for (unsigned int i = 0; i < nrows; i++)
130  for (unsigned int j = 0; j < nrows; j++)
131  (*InvCov)(i, j) = myCorr(i, j);
132  } else {
133  for (unsigned int i = 0; i < nrows; i++)
134  for (unsigned int j = 0; j < nrows; j++)

139  }
140  }
141  } while (!IsEOF);

145  throw std::runtime_error(ss.str());
146  }

153 double HiggsObservable::computeWeight()
154 {
155  double logprob = 0;
156 
157  if (isnew) {
158  for (int i = 0; i < channels.GetNrows(); i++) {
159  double mu = 0;
160  for (unsigned int j = 0; j < thobsvsize; j++) {
161  theoryValues.at(j) = thObsV.at(j)->computeThValue();
162  mu += channels(i, j) * theoryValues.at(j);
163  }
164  theoryValues.at(thobsvsize) = tho->computeThValue();
165  if (thObsV.at(0)->getModel().isModelLinearized())
166  theoryValues.at(thobsvsize + i + 1) = -1. + mu + theoryValues.at(thobsvsize);
167  else
168  theoryValues.at(thobsvsize + i + 1) = mu * theoryValues.at(thobsvsize);
169  if (!isCorrelated) logprob += LogSplitGaussian(theoryValues.at(thobsvsize + i + 1), channels(i, thobsvsize), channels(i, thobsvsize + 1), channels(i, thobsvsize + 2));
170  }
171  if (isCorrelated) {
172  gslpp::vector<double> theObsVal(channels.GetNrows(), 0.);
173  for (int i = 0; i < channels.GetNrows(); i++) {
174  theObsVal(i) = theoryValues.at(thobsvsize + i + 1);
175  }
176  logprob = (-0.5 * theObsVal * ((*InvCov) * theObsVal));
177  }
178  } else {
179  for (int i = 0; i < channels.GetNrows(); i++) {
180  double mu = 0, sum = 0.;
181  for (unsigned int j = 0; j < thobsvsize - 1; j++) {
182  theoryValues.at(j) = thObsV.at(j)->computeThValue();
183  mu += channels(i, j) * theoryValues.at(j);
184  sum += channels(i, j);
185  }
186  mu += (1. - sum) * thObsV.at(thobsvsize - 1)->computeThValue();
187  theoryValues.at(thobsvsize) = tho->computeThValue();
188  theoryValues.at(thobsvsize + i + 1) = mu * theoryValues.at(thobsvsize);
189  logprob += LogSplitGaussian(theoryValues.at(thobsvsize + i + 1), channels(i, 3), channels(i, 4), channels(i, 5));
190  }
191  }
192 
193  return (logprob);
194 }
