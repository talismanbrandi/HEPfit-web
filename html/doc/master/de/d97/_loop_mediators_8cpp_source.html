153 double LoopMediators::F9(double x, double y)
154 {
155  double ym1 = y - 1.;
156  double xm1 = x - 1.;
157 
158  if (x == 1. && y == 1.)
159  return 1./3.;
160  else if (x == 1.)
161  return (-3.*y*y + 2.*y*y*log(y) + 4.*y - 1.)/2./ym1/ym1/ym1;
162  else if (y == 1.)
163  return (-3.*x*x + 2.*x*x*log(x) + 4.*x - 1.)/2./xm1/xm1/xm1;
164  else if (x == y)
165  return (y*y - 2.*y*log(y) - 1.)/ym1/ym1/ym1;
166  else
167  return 1./xm1/ym1 + x*x*log(x)/xm1/xm1/(x-y) + y*y*log(y)/ym1/ym1/(y-x);
168 }
