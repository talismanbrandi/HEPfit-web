59 double HiggsKi::obliqueS() const
60 {
61  double Lambda;
62  if (fabs(1.0 - Kv * Kv) < pow(10.0, -32.0))
63  Lambda = pow(10.0, 19.0);

65  Lambda = 4.0 * M_PI * v() / sqrt(fabs(1.0 - Kv * Kv));

134 double HiggsKi::muVBFpVH(const double sqrt_s) const
135 {
136  double sigmaWH_SM = trueSM.computeSigmaWH(sqrt_s);
137  double sigmaZH_SM = trueSM.computeSigmaZH(sqrt_s);
138  double sigmaWF_SM = trueSM.computeSigmaWF(sqrt_s);
139  double sigmaZF_SM = trueSM.computeSigmaZF(sqrt_s);
140  double sigmaZWF_SM = trueSM.computeSigmaZWF(sqrt_s);
141  double sigmaVBF_SM = sigmaWF_SM + sigmaZF_SM + sigmaZWF_SM;
142 
143  double sigmaWH = muWH(sqrt_s) * sigmaWH_SM;
144  double sigmaZH = muZH(sqrt_s) * sigmaZH_SM;
145  double sigmaVBF = muVBF(sqrt_s) * sigmaVBF_SM;
146 
147  return ((sigmaWH + sigmaZH + sigmaVBF) / (sigmaWH_SM + sigmaZH_SM + sigmaVBF_SM));
148 }
