27 double EW_ABC::Gamma_l(QCD::lepton l) const
28 {
29  double Qf = NPE.getLeptons(l).getCharge();
30  double RQED = 1.0 + 3.0 * NPE.alphaMz() / 4.0 / M_PI * Qf*Qf;
31  double mf = NPE.getLeptons(l).getMass();
32  double beta = sqrt(1.0 - 4.0 * mf * mf / NPE.getMz() / NPE.getMz());
33  double factor = NPE.getGF() * NPE.getMz() * NPE.getMz() * NPE.getMz() / 6.0 / M_PI / sqrt(2.0);
34  return ( factor * beta * ((3.0 - beta * beta) / 2.0 * gVl(l).abs2()
35  + beta * beta * gAl(l).abs2()) * RQED);
36 }

38 double EW_ABC::Gamma_q(QCD::quark q) const
39 {
40  if (q == QCD::BOTTOM || q == QCD::TOP)
41  throw std::runtime_error("Error in EW_ABC::Gamma_q()");
42  double Qf = NPE.getQuarks(q).getCharge();

44  double a = NPE.Als(NPE.getMz(), FULLNNLO) / M_PI;
45  double RQCD = 1.0 + 1.2 * a - 1.1 * a * a - 13.0 * a * a*a;

53  double Nc = 3.0;
54  double factor = NPE.getGF() * NPE.getMz() * NPE.getMz() * NPE.getMz() / 6.0 / M_PI / sqrt(2.0);

61 double EW_ABC::Gamma_b() const
