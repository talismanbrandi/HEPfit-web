451 {

453  double Mb=trueSM.getQuarks(QCD::BOTTOM).getMass();
454  double Mtau=trueSM.getLeptons(StandardModel::TAU).getMass();
455  double MW=trueSM.Mw_tree();
456  double cW2=trueSM.c02();
457  double sW2=1.0-cW2;
458  double TAUt=4.0*Mt*Mt/(mHl*mHl);
459  double TAUb=4.0*Mb*Mb/(mHl*mHl);
460  double TAUtau=4.0*Mtau*Mtau/(mHl*mHl);

463  double LAMb=4.0*Mb*Mb/(Mz*Mz);
464  double LAMtau=4.0*Mtau*Mtau/(Mz*Mz);
465  double LAMw=4.0*MW*MW/(Mz*Mz);
466 
467  double factor = GF*ale*ale/(sqrt(2.0)*64.0*M_PI*M_PI*M_PI)*pow(mHl*(1.0-Mz*Mz/(mHl*mHl)),3);
468 
469  return factor * ((-ct * 4.0*(0.5-4.0/3.0*sW2)*(Int1(TAUt,LAMt)-Int2(TAUt,LAMt)) * (1.0-AlsMz/M_PI)
470  +cb * 2.0*(-0.5+2.0/3.0*sW2)*(Int1(TAUb,LAMb)-Int2(TAUb,LAMb))
471  +ctau * 2.0*(-0.5+2.0*sW2)*(Int1(TAUtau,LAMtau)-Int2(TAUtau,LAMtau)) )/sqrt(sW2*cW2)
