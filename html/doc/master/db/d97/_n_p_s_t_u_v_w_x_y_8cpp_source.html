101 double NPSTUVWXY::obliqueS() const
102 {
103  double sW2_SM = trueSM.sW2();
104  double sW_SM = sqrt(sW2_SM);
105  double cW_SM = trueSM.cW2();
106  return ( (myObliqueShat - myObliqueW + myObliqueX / (sW_SM * cW_SM) - myObliqueY)
107  * 4.0 * sW2_SM / alphaMz());
108 }

110 double NPSTUVWXY::obliqueT() const
111 {
112  double sW2_SM = trueSM.sW2();
113  double sW_SM = sqrt(sW2_SM);
114  double cW2_SM = trueSM.cW2();
115  double cW_SM = sqrt(cW2_SM);
116  return ( (myObliqueThat - myObliqueW + 2.0 * sW_SM / cW_SM * myObliqueX
117  - sW2_SM / cW2_SM * myObliqueY) / alphaMz());
118 }

132 double NPSTUVWXY::GammaW() const
133 {
134  double Gamma_W = trueSM.GammaW();
135 
136  double Wbar = (obliqueV() - obliqueW()) / alphaMz();
137 
138  double alpha = StandardModel::alphaMz();
139  double c2 = trueSM.cW2();
140  double s2 = trueSM.sW2();
141 
142  Gamma_W *= 1.0 - 3.0 * alpha / 4.0 / (c2 - s2)
143  *(obliqueS() - 2.0 * c2 * obliqueT()
144  - (c2 - s2) * obliqueU() / 2.0 / s2 - 2.0 * (c2 - s2) * Wbar)
145  - (1.0 + c2) / 2.0 / (c2 - s2) * DeltaGF();
146 
147  return Gamma_W;
148 }
