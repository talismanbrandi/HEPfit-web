25 double GeneralTHDMSTU::F(const double m02, const double m12) const {
26  double F;
27 
28  if(m02 == 0. && m12 != 0.) {
29  F=0.5 * m12;
30  } else if(m02 != 0. && m12 == 0.){
31  F=0.5 * m02;
32  } else if((m02 == 0. && m12 == 0.) || (fabs(m02-m12) < LEPS)){
33  F=0.;
34  } else if (m02 != 0 && m12 != 0){
35  F=0.5 * (m02 + m12) - (m02 * m12) / (m02 - m12) * log(m02 / m12);
36  } else
37  throw std::runtime_error("Error in GeneralTHDM::F()");
38  return (F);
39 }
