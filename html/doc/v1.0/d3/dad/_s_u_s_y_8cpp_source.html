100 bool SUSY::PostUpdate()
101 {
102  if(!StandardModel::PostUpdate()) return (false);
103  /* Set the squark and slepton mass matrices and the trilinear-coupling matrices */
104  SetSoftTerms();
105  computeYukawas();
106 
107  /* use approximate GUT relation if M1 & M2 are zero */
108  if(m1.abs() == 0. && m2.abs() == 0.) {
109  m1.real() = m3/6.;
110  m2.real() = m3/3.;
111  }
112 
113  /* Compute Higgs and sparticle spectra with FeynHiggs */
114  if (IsFlag_FH()) {
115  }
116  else {
117  /* Compute Higgs and sparticle spectra without FeynHiggs */
118  /* sfermions */
119  for (int i = 0; i < 6; i++) {
120  m_sn2(i) = 0.;// heavy decoupled masses for i=3-5
121  m_se2(i) = 0.;
122  m_su2(i) = 0.;
123  m_sd2(i) = 0.;
124  m_sdresum2(i) = 0.;
125  for (int j = 0; j < 6; j++) {
126  /* R: first (second) index for mass (gauge) eigenstates */
127  /* UASf: second (third) index for gauge (mass) eigenstates */
128  Rn.assign(i,j, 0.);
129  Rl.assign(i,j, 0.);
130  Ru.assign(i,j, 0.);
131  Rd.assign(i,j, 0.);
132  Rdresum.assign(i,j, 0.);
133  }
134  }
135 
136  /* charginos */
137  for (int i = 0; i < 2; i++) {
138  mch(i) = 0.;
139  for (int j = 0; j < 2; j++) {
140  /* U and V: first (second) index for mass (gauge) eigenstates */
141  /* Ucha and VCha: first (second) index for gauge (mass) eigenstates */
142  U.assign(i,j, 0.);
143  V.assign(i,j, 0.);
144  }
145  }
146 
147  /* neutralinos */
148  for (int i = 0; i < 4; i++) {
149  mneu(i) = 0.;
150  for (int j = 0; j < 4; j++)
151  /* N: first (second) index for mass (gauge) eigenstates */
152  /* Zneu: first (second) index for gauge (mass) eigenstates */
153  N.assign(i,j, 0.);
154  }
155 
156  if(!mySUSYSpectrum->CalcHiggs(mh,saeff)) return (false);
157  if(!mySUSYSpectrum->CalcChargino(U,V,mch)) return (false);
158  if(!mySUSYSpectrum->CalcNeutralino(N,mneu)) return (false);
159  if(!mySUSYSpectrum->CalcSup(Ru,m_su2)) return (false);
160  mySUSYSpectrum->SortSfermionMasses(m_su2, Ru);
161  if(!mySUSYSpectrum->CalcSdown(Rd,m_sd2)) return (false);
162  mySUSYSpectrum->SortSfermionMasses(m_sd2, Rd);
163  if(!mySUSYSpectrum->CalcSneutrino(Rn,m_sn2)) return (false);
164  mySUSYSpectrum->SortSfermionMasses(m_sn2, Rn);
165  if(!mySUSYSpectrum->CalcSelectron(Rl,m_se2)) return (false);
166  mySUSYSpectrum->SortSfermionMasses(m_se2, Rl);
167  }
168 
169  /* Set the mass of the SM-like Higgs */
170  /* allowed range for the use of EWSMApproximateFormulae class */
171  if (mh[0] < 10. || mh[0] > 1000.) {
172  std::stringstream out;
173  out << mh[0];
174  throw std::runtime_error("SUSY::PostUpdate(): mh=" + out.str() + " is out of range for EWSMApproximateFormulae");
175  return (false);
176  }
177  mHl = mh[0];
178 
179  if( Q_SUSY == -1 || Q_SUSY == 0) Q_SUSY = sqrt( sqrt(m_su2(2) * m_su2(5)) );
180 
181  /* For EWSUSY class */
182  myEWSUSY->SetRosiekParameters();
183 
184  /* Necessary for updating SUSY and SUSY-derived parameters in SUSYMatching */
185  /* For SUSY Models only: The SM Matching needs to be updated here since StandardModel::PostUpdate() will not do it since the Higgs masses need to be computed. */
186  SUSYM.getObj().updateSMParameters();
187  SUSYM.getObj().updateSUSYParameters();
188 
189  return (true);
190 }

229 void SUSY::SetTanb(const double tanb)
230 {
231  this->tanb = tanb;
232  if (tanb < 0.)
233  throw std::runtime_error("SUSY::setTanb(): Negative tanb is not allowed");
234 
235  /* cosb and sinb are defined to be positive, corresponding to 0<=beta<=pi/2 */
236  cosb = sqrt(1. / (1. + tanb * tanb));
237  sinb = tanb * cosb;
238 }

240 void SUSY::computeYukawas()
241 {
242  /* initializations */
243  Yu = gslpp::matrix<gslpp::complex>::Id(3);
244  Yd = gslpp::matrix<gslpp::complex>::Id(3);
245  Ye = gslpp::matrix<gslpp::complex>::Id(3);
246  Yn = gslpp::matrix<gslpp::complex>::Id(3);
247 
248  /* Convert the top-quark pole mass to the MSbar mass */
249  double mtbar = Mp2Mbar(mtpole, FULLNLO);
250 
251  double Q_SUSY_temp = Q_SUSY;
252  if( Q_SUSY == -1 || Q_SUSY == 0) Q_SUSY = sqrt( sqrt(msQhat2(2,2).abs() * msUhat2(2,2).abs() ) );
253 
254  for (int i = 0; i < 3; i++) {
255  /* Run the quark masses to scale Q */
256  if (i != 2)
257  mu_Q[i] = Mrun(Q_SUSY, getQuarks((quark)(UP + 2 * i)).getMass_scale(),
258  getQuarks((quark)(UP + 2 * i)).getMass(), FULLNLO);
259  else
260  mu_Q[i] = Mrun(Q_SUSY, mtbar, FULLNLO);
261  md_Q[i] = Mrun(Q_SUSY, getQuarks((quark)(DOWN + 2 * i)).getMass_scale(),
262  getQuarks((quark)(DOWN + 2 * i)).getMass(), FULLNLO);
263  me_Q[i] = getLeptons((lepton)(ELECTRON + 2 * i)).getMass();
264  mn_Q[i] = getLeptons((lepton)(NEUTRINO_1 + 2 * i)).getMass();
265 
266  /* Convert MSbar to DRbar */
267  mu_Q[i] = MS2DRqmass(Q_SUSY, mu_Q[i]);
268  md_Q[i] = MS2DRqmass(Q_SUSY, md_Q[i]);
269 
270  Yu.assign(i, i, mu_Q[i] / v2() * sqrt(2.));
271  Yd.assign(i, i, md_Q[i] / v1() * sqrt(2.));
272  Ye.assign(i, i, me_Q[i] / v1() * sqrt(2.));
273  Yn.assign(i, i, mn_Q[i] / v2() * sqrt(2.));
274  }
275 
276  Yu = myCKM.getCKM().transpose()*Yu;
277  Yn = Yn * myPMNS.getPMNS().hconjugate();
278 
279  Q_SUSY = Q_SUSY_temp;
280 
281 }
